Fractal Recursive Tree

This is the Javascript that makes up the Fractal tree using p5.js
There are problems with indenting and HTML, so please copy and paste this code into any IDE and use the auto indent function.
It's all good, you should be able to read it nice and clear.

Code

var length = 200;
var PI = 3.14159265358979323846;
var TWO_PI = PI*2;

var angle = 0;
var slider;

function setup() {
 	createCanvas(400, 400);
  	slider = createSlider(0, TWO_PI, PI / 4, 0.01);
}

function draw() {
  	background(51);
  	angle = slider.value();
  	stroke(255);
  	translate(200, height);
  	branch(100);

}

function branch(len) {
  	line(0, 0, 0, -len);
  	translate(0, -len);
  	if (len > 4) {
    	push();
    		rotate(angle);
    		branch(len * 0.67);
    	pop();

    	push();
    		rotate(-angle);
    		branch(len * 0.67);
    	pop();
  	}
}
